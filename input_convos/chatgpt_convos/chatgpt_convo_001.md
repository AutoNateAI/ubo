        // Simple Load Chart Mockup
        const ctx = document.getElementById('loadChart').getContext('2d');
        let points = [10, 25, 15, 30, 20, 35, 28];
        
        function drawChart() {
            ctx.clearRect(0, 0, 100, 40);
            ctx.beginPath();
            ctx.strokeStyle = '#4edea3';
            ctx.lineWidth = 2;
            ctx.moveTo(0, 40 - points[0]);
            for (let i = 1; i < points.length; i++) {
                ctx.lineTo(i * 16, 40 - points[i]);
            }
            ctx.stroke();
            
            // Shift points for "animation"
            points.shift();
            points.push(Math.random() * 30 + 5);
        }
        
        setInterval(drawChart, 1000);

        // Terminal "Typewriter" effect for new inputs simulation
        const messages = [
            "[14:25:31] Ping: Node Alpha... OK",
            "[14:25:44] Power rail stable.",
            "[14:26:12] Torque sensors calibrated."
        ];
        let msgIdx = 0;
        
        function addTerminalMsg() {
            const terminal = document.querySelector('.bg-black.font-code-sm');
            if (msgIdx < messages.length) {
                const p = document.createElement('p');
                p.className = 'text-outline opacity-0 transition-opacity duration-500';
                p.textContent = messages[msgIdx];
                terminal.insertBefore(p, terminal.lastElementChild);
                setTimeout(() => p.classList.remove('opacity-0'), 100);
                msgIdx++;
            }
        }
        
        setTimeout(addTerminalMsg, 3000);
        setTimeout(addTerminalMsg, 8000);
    </script>

<script>
        const robot = document.getElementById('robot');
        const btns = {
            up: document.getElementById('btn-up'),
            down: document.getElementById('btn-down'),
            left: document.getElementById('btn-left'),
            right: document.getElementById('btn-right')
        };

        let posX = 50;
        let posY = 50;
        let rotation = 0;

        const updateRobot = () => {
            robot.style.left = `${posX}%`;
            robot.style.top = `${posY}%`;
            robot.style.transform = `translate(-50%, -50%) rotate(${rotation}deg)`;
        };

        window.addEventListener('keydown', (e) => {
            const step = 2;
            switch(e.key) {
                case 'ArrowUp':
                    posY = Math.max(10, posY - step);
                    rotation = 0;
                    btns.up.classList.add('joystick-active');
                    break;
                case 'ArrowDown':
                    posY = Math.min(90, posY + step);
                    rotation = 180;
                    btns.down.classList.add('joystick-active');
                    break;
                case 'ArrowLeft':
                    posX = Math.max(10, posX - step);
                    rotation = -90;
                    btns.left.classList.add('joystick-active');
                    break;
                case 'ArrowRight':
                    posX = Math.min(90, posX + step);
                    rotation = 90;
                    btns.right.classList.add('joystick-active');
                    break;
            }
            updateRobot();
        });

        window.addEventListener('keyup', (e) => {
            Object.values(btns).forEach(btn => btn.classList.remove('joystick-active'));
        });

        // Click listeners for on-screen buttons
        btns.up.addEventListener('mousedown', () => { posY = Math.max(10, posY - 5); rotation = 0; updateRobot(); });
        btns.down.addEventListener('mousedown', () => { posY = Math.min(90, posY + 5); rotation = 180; updateRobot(); });
        btns.left.addEventListener('mousedown', () => { posX = Math.max(10, posX - 5); rotation = -90; updateRobot(); });
        btns.right.addEventListener('mousedown', () => { posX = Math.min(90, posX + 5); rotation = 90; updateRobot(); });

    </script>
